Here's a clean implementation using requests and the OpenWeatherMap API. It handles errors gracefully and returns a normalized dict:

Python
import requests def get_weather(city: str, api_key: str) -> dict: """Fetch current weather for a city.""" url = "https://api.openweathermap.org/data/2.5/weather" params = { "q": city, "appid": api_key, "units": "metric" } try: res = requests.get(url, params=params, timeout=5) res.raise_for_status() data = res.json() return { "city": data["name"], "country": data["sys"]["country"], "temp_c": data["main"]["temp"], "feels_like": data["main"]["feels_like"], "humidity": data["main"]["humidity"], "condition": data["weather"][0]["description"], "wind_kph": round(data["wind"]["speed"] * 3.6, 1), } except requests.RequestException as e: return {"error": str(e)}
